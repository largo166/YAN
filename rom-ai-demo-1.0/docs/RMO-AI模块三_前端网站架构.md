## 3.1 整体架构规划

### 3.1.1 多角色系统架构

```
┌─────────────────────────────────────────────────────────────────┐
│                        RMO-AI 平台                               │
│                   (rmo-ai.arch 或类似域名)                        │
├─────────────────┬─────────────────┬─────────────────────────────┤
│    营销端        │    客户端        │        设计师端              │
│   (Landing)     │   (Client)      │      (Designer)            │
├─────────────────┼─────────────────┼─────────────────────────────┤
│ • 首页/产品页    │ • 项目看板       │ • 任务中心                  │
│ • 案例展示       │ • 需求发布       │ • 作品管理                  │
│ • 定价/会员      │ • 进度跟踪       │ • 收益统计                  │
│ • 关于/联系      │ • 成果验收       │ • 技能认证                  │
│                 │ • 发票/合同      │ • 学习社区                  │
├─────────────────┴─────────────────┴─────────────────────────────┤
│                      内部生图系统                                │
│              (Reference DB + Generation Engine)                 │
├─────────────────────────────────────────────────────────────────┤
│                    管理后台（Admin）                             │
│        用户管理 | 任务调度 | 财务结算 | 数据分析 | 内容管理       │
└─────────────────────────────────────────────────────────────────┘
```

### 3.1.2 技术栈建议

```
前端框架：Vue 3 + TypeScript + Vite
UI组件库：Element Plus / Ant Design Vue
状态管理：Pinia
路由：Vue Router 4
HTTP请求：Axios
图表：ECharts
图标：Lucide Vue

后端（配合）：
- API服务：Node.js/Express 或 Python/FastAPI
- 数据库：PostgreSQL（主）+ Redis（缓存）
- 文件存储：OSS/腾讯云COS
- 实时通信：Socket.io / WebSocket

部署：
- 前端：Vercel/Netlify/阿里云OSS
- 后端：阿里云ECS/腾讯云函数
```

### 3.5.3 核心代码实现

```typescript
// services/imageGeneration.ts
import axios from 'axios';

interface GenerateImageParams {
  prompt: string;
  size?: '1024x1024' | '1024x1536' | '1536x1024';
  quality?: 'standard' | 'high';
  style?: 'vivid' | 'natural';
  n?: number;
  referenceImage?: string; // base64 or URL
}

interface GenerationResult {
  id: string;
  urls: string[];
  b64Jsons: string[];
  createdAt: Date;
  prompt: string;
  params: GenerateImageParams;
}

class ImageGenerationService {
  private apiKey: string;
  private baseUrl: string = 'https://api.apiyi.com/v1';
  
  constructor(apiKey: string) {
    this.apiKey = apiKey;
  }

  async generate(params: GenerateImageParams): Promise<GenerationResult> {
    // 优化建筑类prompt
    const optimizedPrompt = this.optimizeArchitecturalPrompt(params.prompt);
    
    const response = await axios.post(
      `${this.baseUrl}/images/generations`,
      {
        model: 'gpt-image-2-all',
        prompt: optimizedPrompt,
        n: params.n || 1,
        size: params.size || '1024x1536',
        quality: params.quality || 'high',
        style: params.style || 'vivid'
      },
      {
        headers: {
          'Authorization': `Bearer ${this.apiKey}`,
          'Content-Type': 'application/json'
        },
        timeout: 120000
      }
    );

    return {
      id: `gen_${Date.now()}`,
      urls: response.data.data.map((d: any) => d.url),
      b64Jsons: response.data.data.map((d: any) => d.b64_json),
      createdAt: new Date(),
      prompt: optimizedPrompt,
      params
    };
  }

  async editImage(
    imagePath: string,
    prompt: string,
    maskPath?: string
  ): Promise<GenerationResult> {
    const formData = new FormData();
    formData.append('image', await this.fileFromPath(imagePath));
    if (maskPath) {
      formData.append('mask', await this.fileFromPath(maskPath));
    }
    formData.append('model', 'gpt-image-2-all');
    formData.append('prompt', prompt);
    formData.append('n', '1');

    const response = await axios.post(
      `${this.baseUrl}/images/edits`,
      formData,
      {
        headers: {
          'Authorization': `Bearer ${this.apiKey}`,
          'Content-Type': 'multipart/form-data'
        }
      }
    );

    return {
      id: `edit_${Date.now()}`,
      urls: response.data.data.map((d: any) => d.url),
      b64Jsons: response.data.data.map((d: any) => d.b64_json),
      createdAt: new Date(),
      prompt,
      params: { prompt, referenceImage: imagePath }
    };
  }

  private optimizeArchitecturalPrompt(prompt: string): string {
    // 检测是否为中文
    const hasChinese = /[\u4e00-\u9fff]/.test(prompt);
    
    const qualityKeywords = `
      architectural visualization, professional rendering,
      high detail, photorealistic, 8k quality,
      clean composition, proper scale, natural lighting
    `.trim();

    if (hasChinese) {
      return `${prompt}\n\n风格: ${qualityKeywords}`;
    } else {
      return `${prompt}, ${qualityKeywords}`;
    }
  }

  private async fileFromPath(path: string): Promise<Blob> {
    // 实现文件读取逻辑
    const response = await fetch(path);
    return response.blob();
  }
}

export default ImageGenerationService;
```

### 3.7.2 任务卡片数据结构

```typescript
// types/workflow.ts

interface TaskCard {
  id: string;
  projectId: string;
  title: string;
  description?: string;
  
  // 任务属性
  type: 'site_plan' | 'unit_layout' | 'facade' | 'rendering' | 'landscape' | 'interior';
  priority: 'P0' | 'P1' | 'P2';
  difficulty: 1 | 2 | 3 | 4 | 5;
  
  // 财务
  budget: number; // 设计师报价
  price: number;  // 客户价格
  
  // 人员
  clientId: string;
  designerId?: string;
  
  // 状态
  status: 'backlog' | 'todo' | 'in_progress' | 'review' | 'revision' | 'done' | 'archived';
  
  // 时间
  createdAt: Date;
  startedAt?: Date;
  deadline: Date;
  completedAt?: Date;
  
  // 进度
  progress: number; // 0-100
  
  // 关联
  dependencies: string[]; // 前置任务ID
  deliverables: Deliverable[];
  comments: Comment[];
  activities: Activity[];
  
  // 标签
  tags: string[];
  
  // AI生成内容
  aiSuggestions?: AISuggestion[];
  generatedImages?: string[];
}

interface Deliverable {
  id: string;
  name: string;
  type: 'file' | 'link' | 'image';
  url: string;
  status: 'draft' | 'submitted' | 'approved' | 'rejected';
  submittedAt?: Date;
  reviewedAt?: Date;
  feedback?: string;
}

interface AISuggestion {
  type: 'task_breakdown' | 'designer_match' | 'risk_alert' | 'time_estimate';
  content: string;
  confidence: number;
  createdAt: Date;
}

// 看板列定义
interface BoardColumn {
  id: string;
  title: string;
  status: TaskCard['status'];
  tasks: TaskCard[];
  wipLimit?: number; // 在制品限制
}

├── 完整的需求发布流程（3步）
├── AI任务拆解集成
├── RMO-AI生图系统（API易接入）
├── 参考图库（基础版）
└── 工作流看板（拖拽功能）

Phase 3: 优化迭代（3-6个月）
├── 智能匹配算法
├── RMO-AI知识库RAG集成
├── 参考图库AI检索
├── 工作流模板库
├── 财务结算自动化
└── 移动端适配
```
